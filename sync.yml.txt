name: News Feed Auto-Engine
on:
  schedule:
    - cron: '*/30 * * * *' # Automatically runs tasks in background every 30 mins
  workflow_dispatch:
permissions:
  contents: write
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v4
    - name: Run Script Engine
      run: python scrape.py
    - name: Commit Live Data
      run: |
        git config --global user.name "GitHub Action Engine"
        git config --global user.email "actions@github.com"
        git add news.json
        git diff-index --quiet HEAD || (git commit -m "Auto Update News File" && git push)
